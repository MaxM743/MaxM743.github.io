Menger Sponge

The Menger Sponge is the generalization in 3D of the Sierpinski Carpet (c.f. first applet). Its construction can be made using the following algorithm:

  1. 1. Start with a cube
  2. 2. Divide the cube into 27 smaller cubes
  3. 3. Remove the cube in the middle and the ones in the center of each face
  4. 4. Repeat the process for each of the remaining cubes

    5. Code example in 2D (p5.js)

            
let maximum = 2;


function setup() {
  createCanvas(600, 600);
  noLoop();
  strokeWeight(0.2);
  stroke('black')
}

function draw() {
  background("black")
  sponge(0, 0, width, 1)
}

function sponge(x, y, length, level){
  if  (level == maximum){
    push()
    fill('blue')
    square(x, y, length);
    pop()
  }else{
    sponge(x, y, length/3, level + 1);
    sponge(x + length/3, y, length/3, level + 1)
    sponge(x, y + length/3, length/3, level + 1)
    sponge(x + 2*length/3, y, length/3, level + 1)
    sponge(x, y + 2*length/3, length/3, level + 1)
    sponge(x + 2*length/3, y + 2*length/3, length/3, level + 1)
    sponge(x + 2*length/3, y + length/3, length/3, level + 1)
    sponge(x + length/3, y + 2*length/3, length/3, level + 1)
  }
}

function mouseClicked(){
  if (maximum < 6){
    maximum++;
    draw()
  }
}
